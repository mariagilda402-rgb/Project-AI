      function connectWS() {
        try {
          ws = new WebSocket('ws://127.0.0.1:5123/ws');
          ws.onmessage = (e) => {
            try {
              const d = JSON.parse(e.data);
              if (d.type === 'state' && d.status) {
                const lbl = document.getElementById('statusLabel');
                lbl.textContent = STATE_LABELS[d.status] || d.status.toUpperCase();
                lbl.setAttribute('data-state', d.status);
              }
            } catch(ex) {}
          };
          ws.onclose = () => setTimeout(connectWS, 3000);
          ws.onerror = () => {};
        } catch(e) { setTimeout(connectWS, 3000); }
      }

      async function loadMemories() {
        const collection = document.getElementById('memAgentSelect').value;
        const query = document.getElementById('memSearch').value.trim();
        if (!collection) return;
        const result = await apiCall('list_memories', collection, query);
        const list = document.getElementById('memList');
        const count = document.getElementById('memCount');
        if (!result) { list.innerHTML = '<div class="memory-empty">Erro ao carregar memÃ³rias.</div>'; return; }
        count.textContent = `${result.total} memÃ³ria(s) encontrada(s)`;
        if (result.memories.length === 0) {
          list.innerHTML = '<div class="memory-empty">Nenhuma memÃ³ria encontrada.</div>';
          return;
        }
        list.innerHTML = '';
        result.memories.forEach(m => {
          const div = document.createElement('div');
          div.className = 'memory-item';
          div.innerHTML = `
            <div class="memory-text">${m.text}</div>
